$ Z.z.: Berechne (-1 + i)^{10} $

\begin{align*}
	\\
	(-1 + i)^{10}
	\\
    = ((-1 + i)^2)^5 \text{Potenzregeln}\\
	\\
	=(1 -2i -1)^5 \text{quadrat rein multipliziert}\\
	\\
	=(-2i)^5 \text{ausgerechnet}\\
	\\
	= -32i^5
	\\
	= -32 * i^2 + i^2 +i \text{Potenzregeln}\\
	\\
	= -32 * (-1) * (-1) *i
	\\
	= -32i
\end{align*}
